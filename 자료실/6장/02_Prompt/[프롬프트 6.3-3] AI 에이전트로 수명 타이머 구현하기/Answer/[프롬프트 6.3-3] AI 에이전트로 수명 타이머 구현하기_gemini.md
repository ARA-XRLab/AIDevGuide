    3 void AEnemyBulletActor::SetActive(bool bActive)
    4 {
    5     bIsActive = bActive;
    6     MeshComp->SetVisibility(bActive);
    7     BoxComp->SetCollisionEnabled(bActive ? ECollisionEnabled::Type::QueryAndPhysics : ECollisionEnabled::Type::NoCollision);
    8
    9     if (bActive)
   10     {
   11         // SetActive(true) 시점에 LifeTime 기반 수명 타이머 시작
   12         GetWorldTimerManager().SetTimer(LifeTimeTimerHandle, this, &AEnemyBulletActor::OnLifeTimeExpired, LifeTime, false);
   13     }
   14     else
   15     {
   16         GetWorldTimerManager().ClearTimer(LifeTimeTimerHandle);
   17     }
   18 }


  기존 코드에 이미 LifeTimeTimerHandle 변수와 OnLifeTimeExpired 함수(기본 구현: Destroy())가 정의되어 있어, 이를 활용하여 요청하신 로직을 구현했습니다.
